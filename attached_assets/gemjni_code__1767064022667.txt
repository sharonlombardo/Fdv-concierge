// --- PURE LOCAL STORAGE ENGINE (Crash-Proof) ---
const useJournal = () => {
  const [entries, setEntries] = useState({});
  const [status, setStatus] = useState('idle');

  useEffect(() => {
    try {
      const saved = localStorage.getItem('morocco_soul_final_v15');
      if (saved) {
        setEntries(JSON.parse(saved));
      }
    } catch (e) {
      console.warn("Storage restricted");
    }
  }, []);

  const saveEntry = (id, data) => {
    setStatus('saving');
    setTimeout(() => {
      const currentEntry = entries[id] || {};
      const updatedEntries = {
        ...entries,
        [id]: { ...currentEntry, ...data, updatedAt: Date.now() }
      };
      setEntries(updatedEntries);
      try {
        localStorage.setItem('morocco_soul_final_v15', JSON.stringify(updatedEntries));
      } catch (e) { console.warn("Quota exceeded"); }
      setStatus('saved');
      
      // Reset status back to idle after a moment
      setTimeout(() => setStatus('idle'), 2000);
    }, 400);
  };

  return { entries, saveEntry, status };
};

const WeatherDisplay = ({ weather }) => {
  if (!weather) return null;
  const getIcon = (cond) => {
    const c = cond?.toLowerCase() || '';
    if (c.includes('rain')) return <CloudRain size={18} />;
    if (c.includes('wind')) return <Wind size={18} />;
    if (c.includes('cloud')) return <Cloud size={18} />;
    return <Sun size={18} />;
  };
  return (
    <div className="flex items-center gap-4 bg-white/50 backdrop-blur px-5 py-2.5 rounded-full border border-black/10 shadow-sm">
      <div className="text-black">{getIcon(weather.cond)}</div>
      <div className="flex flex-col">
        <span className="text-[11px] font-black uppercase tracking-tighter leading-none">{weather.temp}°F / {weather.cond}</span>
        <span className="text-[9px] text-slate-500 uppercase tracking-widest leading-none mt-1">Live Conditions</span>
      </div>
    </div>
  );
};

// --- ERROR BOUNDARY ---
class ErrorBoundary extends React.Component {
  constructor(props) {
    super(props);
    this.state = { hasError: false, error: null };
  }
  static getDerivedStateFromError(error) {
    return { hasError: true, error };
  }
  render() {
    if (this.state.hasError) {
      return (
        <div className="fixed inset-0 flex items-center justify-center bg-white p-8 text-center">
          <div>
            <h2 className="text-xl font-bold mb-2">Something went wrong</h2>
            <pre className="text-xs bg-gray-100 p-4 rounded text-left overflow-auto max-w-sm">{this.state.error?.toString()}</pre>
            <button onClick={() => window.location.reload()} className="mt-4 px-4 py-2 bg-black text-white rounded">Reload</button>
          </div>
        </div>
      );
    }
    return this.props.children;
  }
}

      const img = new Image();
      img.src = event.target.result;
      img.onload = () => {
        const canvas = document.createElement('canvas');
        const MAX_WIDTH = 1200;
        let width = img.width;
        let height = img.height;
        if (width > MAX_WIDTH) { height *= MAX_WIDTH / width; width = MAX_WIDTH; }
        canvas.width = width; canvas.height = height;
        const ctx = canvas.getContext('2d');
        ctx.drawImage(img, 0, 0, width, height);
        const base64 = canvas.toDataURL('image/jpeg', 0.6);
        saveEntry(itemId, { [field]: base64 });

              <div className="flex flex-col md:flex-row md:items-end justify-between gap-10 mb-20 border-b border-black/10 pb-16">
                <div className="space-y-6">
                  <p className="text-[12px] font-black tracking-[0.5em] uppercase text-slate-400">DAY {currentPage.day} — {currentPage.date}</p>
                  <h2 className="text-6xl md:text-8xl font-black tracking-tighter leading-none m-0 uppercase">{currentPage.title}</h2>
                  <div className="flex items-center gap-4 text-[12px] font-black uppercase tracking-[0.4em] text-slate-500"><MapPin size={16} className="text-black" /> {currentPage.location}</div>
                </div>
                <WeatherDisplay weather={currentPage.weather} />
              </div>

              <div className="mb-20 space-y-6">
                  <h3 className="text-[11px] font-black uppercase tracking-[0.6em] flex items-center gap-3"><Info size={16} /> FIELD NOTES</h3>
                  <div className="text-2xl leading-relaxed text-slate-800 font-serif italic border-l-2 border-black/10 pl-10 opacity-90 whitespace-pre-wrap">{currentPage.fieldNotes}</div>
              </div>

              <div className="mb-32">
                <h3 className="text-[11px] font-black uppercase tracking-[0.6em] pb-4 border-b-2 border-black mb-16">SCHEDULE</h3>
                <div className="space-y-6">
                  {currentPage.flow.map((item, i) => (
                    <button key={i} onClick={() => setActiveItem(item)} className="group w-full text-left p-10 bg-white border border-black/5 hover:border-black transition-all rounded-[2rem] flex gap-10 items-start shadow-sm active:scale-[0.98]">
                      <div className="text-[12px] font-black tracking-widest text-slate-200 group-hover:text-black transition-colors w-24 pt-1 shrink-0">{item.time}</div>
                      <div className="flex-1 space-y-4">
                        <div className="text-[10px] font-black tracking-[0.4em] uppercase text-slate-400 group-hover:text-black">{item.heading}</div>
                        <div className="flex items-center gap-4">
                          <h4 className="text-3xl font-black tracking-tight leading-none group-hover:opacity-70">{item.title}</h4>
                          {journalEntries[item.id] && <div className="flex items-center gap-1 text-[9px] font-black uppercase tracking-tighter bg-black text-white px-2 py-0.5 rounded-full"><CloudUpload size={10} /> Logged</div>}
                        </div>
                        <p className="text-lg text-slate-500 line-clamp-2 italic font-serif leading-relaxed opacity-60 group-hover:opacity-100 transition-all">{item.description || item.body}</p>
                      </div>
                      <ArrowRight size={24} className="text-slate-100 group-hover:text-black group-hover:translate-x-3 transition-all shrink-0" />
                    </button>
                  ))}
                </div>
              </div>

              <div className="pt-20 border-t-2 border-black/10 flex flex-col items-center">
                  <h3 className="text-[11px] font-black uppercase tracking-[0.6em] mb-12">DAILY MANTRA</h3>
                  <p className="text-2xl md:text-3xl font-black tracking-widest uppercase italic text-center max-w-2xl leading-relaxed">"{currentPage.mantra}"</p>
              </div>
            </div>
          )}

          {/* JOURNAL */}
          {currentPage.type === 'journal' && (
            <div className="flex-1 py-12 animate-in fade-in duration-1000">
              <div className="mb-24 text-center">
                <h2 className="text-7xl font-black tracking-tighter mb-6 uppercase">{currentPage.title}</h2>
                <div className="h-1.5 w-24 bg-black mx-auto" />
                <p className="text-sm tracking-[0.6em] font-black uppercase text-slate-400 mt-12">{currentPage.subtitle}</p>
              </div>
              <div className="space-y-32">
                {Object.keys(journalEntries).length > 0 ? Object.keys(journalEntries).sort().map((key) => {
                    const entry = journalEntries[key];
                    if (!entry.note && !entry.image) return null;
                    return (
                      <div key={key} className="grid md:grid-cols-12 gap-16 items-center group">
                        <div className="md:col-span-7 aspect-square grayscale group-hover:grayscale-0 transition-all duration-1000 shadow-xl overflow-hidden bg-black/5">
                          <img src={entry.image || "https://images.unsplash.com/photo-1510414842594-a61c69b5ae57?auto=format&fit=crop&q=80&w=800"} className="w-full h-full object-cover" />
                        </div>
                        <div className="md:col-span-5 space-y-6">
                          <div className="text-[10px] font-black tracking-[0.4em] uppercase opacity-30 border-b border-black/10 pb-2">REF: {key.toUpperCase()}</div>
                          <p className="text-2xl md:text-3xl leading-[1.6] font-serif italic opacity-90">"{entry.note || "A moment captured."}"</p>
                        </div>
                      </div>
                    );
                  }) : <div className="text-center py-20 border-2 border-dashed border-black/10 rounded-3xl opacity-30 italic uppercase tracking-widest text-[10px]">Memories will persist here once captured.</div>}
              </div>
            </div>
          )}

          <footer className="mt-32 border-t border-black/10 pt-16 flex items-center justify-between">
            <button onClick={prevPage} disabled={pageIndex === 0} className="flex items-center gap-4 text-[12px] font-black uppercase tracking-[0.5em] disabled:opacity-0 transition-all hover:-translate-x-4 group"><ChevronLeft size={24} className="group-hover:text-black" /> PREV</button>
            <div className="text-[11px] font-black tracking-[0.8em] text-slate-300 uppercase">{pageIndex + 1} / {ITINERARY_DATA.length}</div>
            <button onClick={nextPage} disabled={pageIndex === ITINERARY_DATA.length - 1} className="flex items-center gap-4 text-[12px] font-black uppercase tracking-[0.5em] disabled:opacity-0 transition-all hover:translate-x-4 group">NEXT <ChevronRight size={24} className="group-hover:text-black" /></button>
          </footer>
        </div>

        {/* NAV */}
        <nav className="fixed top-0 left-0 right-0 p-6 flex justify-between items-center z-[50] bg-[#F5F5F3]/80 backdrop-blur-md border-b border-black/5">
          <div className="text-[11px] font-black uppercase tracking-[0.5em] cursor-pointer" onClick={() => setPageIndex(0)}>MOROCCO / 2026</div>
          <button onClick={() => setMenuOpen(!menuOpen)} className="text-[11px] font-black uppercase tracking-[0.2em] flex items-center gap-3 group">
            <div className="w-10 h-10 flex items-center justify-center rounded-full bg-black/5 group-hover:bg-black group-hover:text-white transition-all">{menuOpen ? <X size={16} /> : <Menu size={16} />}</div>
            {menuOpen ? 'CLOSE' : 'INDEX'}
          </button>
        </nav>

        {/* MENU */}
        {menuOpen && (
          <div className="fixed inset-0 bg-black text-[#F5F5F3] z-[200] flex flex-col justify-center items-center p-8 transition-all duration-500 overflow-y-auto">
            <div className="w-full max-w-lg space-y-8 text-center py-24">
              {ITINERARY_DATA.map((p, i) => (
                <button key={i} className={`block w-full text-2xl md:text-3xl uppercase tracking-[0.2em] transition-all transform hover:scale-105 ${pageIndex === i ? 'opacity-100 font-black italic underline underline-offset-8' : 'opacity-25 hover:opacity-100'}`} onClick={() => { setPageIndex(i); setMenuOpen(false); }}>
                  {p.type === 'cover' ? 'Start' : p.day ? `Day ${p.day}: ${p.title.replace('Daily Flow: ', '')}` : p.title}
                </button>
              ))}
            </div>
          </div>
        )}

        {/* MODALS - MOVED TO BOTTOM FOR Z-INDEX SAFETY */}
        
        {/* Instagram Preview */}
        {isShareMode && activeItem && (
          <div className="fixed inset-0 z-[9999] bg-black flex flex-col items-center justify-center p-4">
            <button onClick={() => setIsShareMode(false)} className="fixed top-8 right-8 text-white/50 hover:text-white transition-colors z-[10000]"><X size={32} /></button>
            <div id="story-frame" className="w-full max-w-[400px] aspect-[9/16] bg-[#F5F5F3] shadow-2xl flex flex-col justify-between p-12 relative overflow-hidden animate-in fade-in zoom-in-95 duration-700">
              <div className="space-y-8 z-10">
                <div className="text-[12px] font-black tracking-[0.6em] uppercase opacity-40">MOROCCO — 2026</div>
                <div className="aspect-[4/5] w-full overflow-hidden grayscale bg-black shadow-2xl">
                  <img src={entries[activeItem.id]?.image || activeItem.image} className="w-full h-full object-cover" alt="Moment" />
                </div>
              </div>
              <div className="space-y-8 z-10 mt-auto">
                <h4 className="text-4xl font-black tracking-tighter leading-none uppercase">{activeItem.title}</h4>
                <p className="text-xl italic opacity-80 leading-relaxed font-serif">"{entries[activeItem.id]?.note || "A rhythm found in the Red City."}"</p>
                <div className="flex justify-between items-end pt-8 border-t border-black/5 text-[9px] font-black tracking-[0.4em] opacity-30">
                  <span>SOUL OF THE MAGHREB</span>
                  <MapPin size={14} className="opacity-20" />
                </div>
              </div>
              <div className="absolute inset-0 pointer-events-none opacity-[0.03] bg-[url('https://www.transparenttextures.com/patterns/natural-paper.png')]" />
            </div>

            <div className="mt-12 flex flex-col items-center gap-6">
              <button onClick={handleNativeShare} className="flex items-center gap-4 text-[12px] font-black uppercase tracking-[0.4em] bg-white text-black px-12 py-5 rounded-full active:scale-95 transition-all shadow-2xl hover:bg-white/90">
                <Instagram size={20} /> Share / Save
              </button>
              <p className="text-[10px] font-black uppercase tracking-[0.2em] text-white/40 italic">Screenshot if menu fails</p>
            </div>
          </div>
        )}

        {/* Detailed Item Drawer (Reordered 1-10) */}
        {activeItem && !isShareMode && (
          <div className="fixed inset-0 z-[9999] bg-[#F5F5F3] p-6 md:p-12 overflow-y-auto animate-in slide-in-from-right duration-500">
            <div className="fixed top-6 right-6 z-[10000] flex gap-4">
               <button onClick={() => setActiveItem(null)} className="px-6 py-2 bg-black text-white text-[10px] font-black uppercase tracking-[0.2em] rounded-full shadow-2xl hover:opacity-80 transition-opacity">Close</button>
            </div>
            
            <div className="max-w-2xl mx-auto space-y-16 pb-32 pt-12">
              
              {/* 1. WEATHER & TIME */}
              <div className="flex items-center justify-between border-b border-black/10 pb-6">
                  <div className="flex items-center gap-3 text-[11px] font-black tracking-[0.4em] uppercase text-slate-400"><Clock size={14} /> {activeItem.time}</div>
                  <div className="flex items-center gap-2 text-[11px] font-black uppercase tracking-[0.2em]"><Sun size={14} /> 72°F</div>
              </div>

              {/* 2. DESCRIPTION */}
              <div className="space-y-4">
                <h2 className="text-5xl font-black tracking-tighter leading-none">{activeItem.title}</h2>
                <div className="aspect-[16/9] w-full overflow-hidden grayscale hover:grayscale-0 transition-all duration-1000 bg-black/5 shadow-inner my-6">
                  <img src={activeItem.image} className="w-full h-full object-cover" />
                </div>
                <p className="text-xl text-slate-800 leading-[1.8] font-serif italic border-l-2 border-black/10 pl-8">{activeItem.description}</p>
              </div>

              {/* 3. LOGISTICS */}
              <div className="pt-8 border-t border-black/10">
                <h3 className="text-[11px] font-black tracking-[0.5em] uppercase mb-8 flex items-center gap-3"><MapPin size={16} /> LOGISTICS</h3>
                <div className="space-y-4">
                  {activeItem.contact && <a href={`tel:${activeItem.contact}`} className="flex items-center gap-4 text-sm font-bold border-b border-black/5 pb-3 hover:border-black transition-all"><Phone size={14} /> {activeItem.contact}</a>}
                  {activeItem.map && <a href={activeItem.map} target="_blank" className="flex items-center gap-4 text-sm font-bold border-b border-black/5 pb-3 hover:border-black transition-all"><Navigation size={14} /> Open in Maps</a>}
                  {activeItem.ticketLink && (
                    <a href={activeItem.ticketLink} target="_blank" className="flex items-center gap-4 text-sm font-bold bg-black text-white px-6 py-4 rounded-full w-fit hover:opacity-80 transition-all">
                      <Ticket size={14} /> Book Tickets
                    </a>
                  )}
                </div>
              </div>

              {/* 4-7. WARDROBE (Commercial & User Swap) */}
              <div className="pt-16 border-t border-black/10">
                <h3 className="text-[11px] font-black tracking-[0.5em] uppercase mb-8 flex items-center gap-3"><Sparkles size={16} /> THE LOOK</h3>
                <p className="text-base font-medium italic opacity-70 leading-relaxed mb-8">"{activeItem.wardrobe}"</p>
                
                <div className="grid md:grid-cols-2 gap-8">
                  {/* Fil de Vie Suggestion */}
                  <div className="space-y-4">
                     <div className="aspect-[3/4] w-full bg-black/5 overflow-hidden grayscale">
                        <img src={activeItem.commercialWardrobe || "https://images.unsplash.com/photo-1594633312681-425c7b97ccd1?auto=format&fit=crop&q=80&w=800"} className="w-full h-full object-cover" />
                     </div>
                     <div className="flex justify-between items-center">
                       <p className="text-[10px] font-black uppercase tracking-widest opacity-40 italic">Fil de Vie Recommendation</p>
                       <button className="text-[10px] font-bold underline">Shop Look</button>
                     </div>
                  </div>

                  {/* User Swap */}
                  <div className="space-y-4">
                     <label className="aspect-[3/4] w-full bg-white border-2 border-dashed border-black/10 flex flex-col items-center justify-center gap-4 hover:bg-black/5 cursor-pointer relative overflow-hidden group">
                        <input type="file" accept="image/*" className="absolute inset-0 w-full h-full opacity-0 cursor-pointer z-10" onChange={(e) => processImage(activeItem.id, e.target.files[0], 'myLook')} />
                        {entries[activeItem.id]?.myLook ? (
                          <img src={entries[activeItem.id].myLook} className="w-full h-full object-cover" />
                        ) : (
                          <>
                            <ShoppingBag size={24} className="opacity-20 group-hover:opacity-100 transition-opacity" />
                            <span className="text-[10px] font-black uppercase tracking-[0.2em] text-center px-4">Upload My Look</span>
                          </>
                        )}
                     </label>
                     <p className="text-[10px] font-black uppercase tracking-widest opacity-40 italic text-center">Tap to Swap</p>
                  </div>
                </div>
              </div>

              {/* 8-9. TRAVEL LOG (Text & Photo) */}
              <div className="pt-16 border-t border-black/10">
                 <h3 className="text-[11px] font-black tracking-[0.5em] uppercase mb-8 flex items-center gap-3"><PenLine size={16} /> TRAVEL LOG</h3>
                 
                 <textarea 
                  placeholder="Record the moment..."
                  className="w-full bg-white border border-black/5 p-6 font-serif italic text-lg focus:outline-none focus:border-black/20 transition-all min-h-[150px] rounded-xl mb-8"
                  value={entries[activeItem.id]?.note || ''}
                  onChange={(e) => handleJournalChange(activeItem.id, e.target.value)}
                />

                <div className="grid md:grid-cols-2 gap-8 items-start">
                   <label className="aspect-square bg-white border-2 border-dashed border-black/10 flex flex-col items-center justify-center gap-4 hover:bg-black/5 cursor-pointer relative overflow-hidden group rounded-xl">
                      <input type="file" accept="image/*" className="absolute inset-0 w-full h-full opacity-0 cursor-pointer z-10" onChange={(e) => processImage(activeItem.id, e.target.files[0], 'logImage')} />
                      {entries[activeItem.id]?.logImage ? (
                        <img src={entries[activeItem.id].logImage} className="w-full h-full object-cover grayscale" />
                      ) : (
                        <>
                          <Camera size={32} className="opacity-20 group-hover:opacity-100 transition-opacity" />
                          <span className="text-[10px] font-black uppercase tracking-[0.2em]">Upload Photo</span>
                        </>
                      )}
                   </label>

                   <div className="space-y-6">
                      {/* 9. SHARE BUTTON */}
                      <button onClick={() => setIsShareMode(true)} className="w-full flex items-center justify-center gap-3 bg-black text-white px-6 py-4 rounded-lg hover:opacity-80 transition-opacity">
                         <Share2 size={14} />
                         <span className="text-[10px] font-black uppercase tracking-widest">Share to Instagram</span>
                      </button>

                      {/* 10. SAVE BUTTON */}
                      <div className="flex items-center justify-center gap-2 p-4 bg-black/5 rounded-lg">
                         {status === 'saving' ? <Loader2 size={14} className="animate-spin" /> : <Save size={14} />}
                         <span className="text-[10px] font-black uppercase tracking-widest">{status === 'saving' ? 'Saving...' : 'Saved to Log'}</span>
                      </div>
                   </div>
                </div>
              </div>

            </div>
          </div>
        )}
      </StoneBackground>
    </ErrorBoundary>
  );
}
